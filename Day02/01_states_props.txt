# what are probs?
probs = data passesd from parent component -> child component
        they are read-only (child cannot change them)   

# what are states?
state = data managed inside a component 
        it can change using setState(or useState in functional components)

STEP 5: Add State in Student Component
--------------------------------------
Now update Student.js
import { useState } from "react";

function Student(props) {

  const [count, setCount] = useState(0);

  return (
    <div>
      <h2>Student Component</h2>

      <p>Name: {props.name}</p>
      <p>Course: {props.course}</p>

      <p>Clicks: {count}</p>

      <button onClick={() => setCount(count + 1)}>
        Increase
      </button>
    </div>
  );
}

export default Student;
